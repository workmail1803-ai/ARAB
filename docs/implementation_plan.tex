\section{Executive Summary}
% ============================================

This document outlines the technical implementation plan for building a comprehensive \textbf{B2B E-commerce \& Fleet Management SaaS Platform}. The platform enables companies to manage their entire delivery operations including:

\begin{itemize}[leftmargin=*]
    \item \textbf{Order Management} - View, track, and manage all orders
    \item \textbf{Fleet Tracking} - Real-time rider/driver location monitoring
    \item \textbf{Analytics \& Reports} - Revenue, orders, and performance metrics
    \item \textbf{Customer Management} - Customer profiles and order history
    \item \textbf{Inventory Tracking} - Stock levels and product management
    \item \textbf{Payment Processing} - Transaction history and revenue tracking
    \item \textbf{API Integration} - Connect existing systems via REST API
\end{itemize}

\subsection{Complete Schema}

\begin{lstlisting}[language=SQL, caption={Complete database schema}]
-- Companies (Tenants)
CREATE TABLE companies (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    api_key VARCHAR(64) UNIQUE NOT NULL,
    webhook_secret VARCHAR(64) NOT NULL,
    plan VARCHAR(50) DEFAULT 'free',
    settings JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT NOW()
);

-- Riders/Agents
CREATE TABLE riders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    company_id UUID REFERENCES companies(id),
    name VARCHAR(255) NOT NULL,
    phone VARCHAR(50),
    email VARCHAR(255),
    status VARCHAR(20) DEFAULT 'offline',
    latitude DECIMAL(10, 8),
    longitude DECIMAL(11, 8),
    battery_level INTEGER,
    vehicle_type VARCHAR(50),
    last_seen TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Customers
CREATE TABLE customers (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    company_id UUID REFERENCES companies(id),
    external_id VARCHAR(255),
    name VARCHAR(255) NOT NULL,
    phone VARCHAR(50),
    email VARCHAR(255),
    addresses JSONB DEFAULT '[]',
    total_orders INTEGER DEFAULT 0,
    total_spent DECIMAL(12, 2) DEFAULT 0,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Products/Inventory
CREATE TABLE products (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    company_id UUID REFERENCES companies(id),
    sku VARCHAR(100),
    name VARCHAR(255) NOT NULL,
    description TEXT,
    price DECIMAL(10, 2),
    stock_quantity INTEGER DEFAULT 0,
    low_stock_threshold INTEGER DEFAULT 10,
    category VARCHAR(100),
    image_url TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Orders
CREATE TABLE orders (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    company_id UUID REFERENCES companies(id),
    external_id VARCHAR(255),
    customer_id UUID REFERENCES customers(id),
    rider_id UUID REFERENCES riders(id),
    status VARCHAR(30) DEFAULT 'pending',
    pickup_address TEXT,
    delivery_address TEXT NOT NULL,
    items JSONB DEFAULT '[]',
    subtotal DECIMAL(10, 2),
    delivery_fee DECIMAL(10, 2),
    total DECIMAL(10, 2),
    payment_method VARCHAR(50),
    payment_status VARCHAR(30) DEFAULT 'pending',
    notes TEXT,
    scheduled_at TIMESTAMP,
    picked_up_at TIMESTAMP,
    delivered_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT NOW()
);

-- Transactions
CREATE TABLE transactions (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    company_id UUID REFERENCES companies(id),
    order_id UUID REFERENCES orders(id),
    type VARCHAR(30),  -- 'payment', 'refund', 'cod_collection'
    amount DECIMAL(10, 2),
    status VARCHAR(30),
    payment_method VARCHAR(50),
    rider_id UUID REFERENCES riders(id),
    created_at TIMESTAMP DEFAULT NOW()
);

-- Location History (for route tracking)
CREATE TABLE location_history (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    rider_id UUID REFERENCES riders(id),
    latitude DECIMAL(10, 8),
    longitude DECIMAL(11, 8),
    speed DECIMAL(5, 2),
    heading INTEGER,
    recorded_at TIMESTAMP DEFAULT NOW()
);
\end{lstlisting}

\subsection{WordPress WooCommerce Integration}

\begin{lstlisting}[language=PHP, caption={WooCommerce integration}]
<?php
// Add to functions.php

// Send new orders to fleet platform
add_action('woocommerce_new_order', 'sync_order_to_fleet');

function sync_order_to_fleet($order_id) {
    $order = wc_get_order($order_id);
    $api_key = get_option('fleet_api_key');
    
    $payload = array(
        'external_id' => $order_id,
        'customer_name' => $order->get_billing_first_name(),
        'customer_phone' => $order->get_billing_phone(),
        'delivery_address' => $order->get_shipping_address_1(),
        'total' => $order->get_total(),
        'items' => array()
    );
    
    foreach ($order->get_items() as $item) {
        $payload['items'][] = array(
            'name' => $item->get_name(),
            'quantity' => $item->get_quantity(),
            'price' => $item->get_total()
        );
    }
    
    wp_remote_post('https://your-platform.com/api/v1/orders', array(
        'headers' => array(
            'Authorization' => 'Bearer ' . $api_key,
            'Content-Type' => 'application/json'
        ),
        'body' => json_encode($payload)
    ));
}
?>
\end{lstlisting}

\subsection{JavaScript/Node.js Integration}

\begin{lstlisting}[language=JavaScript, caption={Node.js integration}]
const axios = require('axios');

class FleetAPI {
  constructor(apiKey) {
    this.client = axios.create({
      baseURL: 'https://your-platform.com/api/v1',
      headers: { 'Authorization': `Bearer ${apiKey}` }
    });
  }

  // Orders
  async createOrder(orderData) {
    return this.client.post('/orders', orderData);
  }

  async getOrders(filters = {}) {
    return this.client.get('/orders', { params: filters });
  }

  // Riders
  async updateRiderLocation(riderId, lat, lng) {
    return this.client.patch(`/riders/${riderId}/location`, {
      latitude: lat,
      longitude: lng
    });
  }

  // Analytics
  async getDashboardKPIs(period = 'today') {
    return this.client.get('/analytics/kpis', { 
      params: { period } 
    });
  }

  // Inventory
  async syncInventory(products) {
    return this.client.post('/products/sync', { products });
  }
}

// Usage
const fleet = new FleetAPI('your-api-key');
await fleet.createOrder({
  customer_name: 'Ahmed',
  delivery_address: '123 Main St, Riyadh',
  items: [{ product_id: 'xxx', quantity: 2 }]
});
\end{lstlisting}
